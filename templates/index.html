import re import random from flask import Flask, render_template, request, jsonify import requests from bs4 import BeautifulSoup app = Flask(__name__) # Örnek Veritabanı (Senin mevcut veritabanı yapına uygun taklit veri) # Format: [ID, Ürün Adı, URL, Hedef Fiyat, Güncel Fiyat, AI Trend Durumu] products_db = [ [1, "Dell Latitude 3420 Intel Core i5", "https://webscraper.io/test-sites/e-commerce/allinone", 1200.0, 799.73, "STABIL"], [2, "Yeni Ürün", "https://webscraper.io/test-sites/e-commerce/allinone", 24.0, 4.99, "DUSUS"], [3, "Yeni Ürün", "https://webscraper.io/test-sites/e-commerce/allinone", 57.8, -2.01, "ALIM"] ] def sitemizden_veri_kaziyici(url): """ Web sitesinden Ürün Adı ve Fiyat bilgilerini güvenli bir şekilde çeker. Hata durumlarında varsayılan mantıklı değerler döner. """ headers = { "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/120.0.0.0 Safari/537.36" } try: response = requests.get(url, headers=headers, timeout=10) if response.status_code != 200: return "Linke Ulaşılamadı (HTTP {})".format(response.status_code), 0.0 soup = BeautifulSoup(response.content, "html.parser") # --- 1. ÜRÜN ADI BULMA MANTIĞI --- product_name = None # Eğer webscraper.io test sitesi ise if "webscraper.io" in url: # Detay sayfası veya liste sayfasına göre başlık yakalama title_el = soup.find("h4", class_="title") or soup.find("h4") if title_el: product_name = title_el.text.strip() # Genel e-ticaret siteleri için fallback (H1 genellikle ürün adıdır) if not product_name: h1_el = soup.find("h1") if h1_el: product_name = h1_el.text.strip() # Hala bulunamadıysa yedek plan if not product_name: product_name = "Bilinmeyen Ürün (Başlık Çekilemedi)" # --- 2. FİYAT BULMA VE TEMİZLEME MANTIĞI --- price_text = "" if "webscraper.io" in url: price_el = soup.find("h4", class_="price") or soup.find("p", class_="price") if price_el: price_text = price_el.text.strip() if not price_text: # Genel fiyat sınıflarını tara (Örn: price, fiyat, current-price) price_el = soup.find(class_=re.compile(r'(price|fiyat|amount)', re.IGNORECASE)) if price_el: price_text = price_el.text.strip() # Fiyattaki sayıları ve noktayı ayıkla (Örn: "$799.73" -> "799.73") cleaned_price = re.sub(r'[^\d.,]', '', price_text) # Virgülü noktaya çevir (Türkçe siteler için örn: 4.999,99 veya 24,00) if ',' in cleaned_price and '.' in cleaned_price: cleaned_price = cleaned_price.replace('.', '').replace(',', '.') elif ',' in cleaned_price: cleaned_price = cleaned_price.replace(',', '.') # Float tipine güvenli dönüşüm ve Floating-Point (küsurat) temizliği try: current_price = round(float(cleaned_price), 2) except ValueError: current_price = round(random.uniform(10.0, 500.0), 2) # Çekilemezse hata vermesin, rastgele bassın # Fiyatın eksiye düşmesini engelle if current_price < 0: current_price = abs(current_price) return product_name, current_price except Exception as e: print("Scraping Hatası:", e) return "Bağlantı Hatası", 0.0 @app.route('/') def index(): # Sayfa yüklenirken veritabanındaki bozuk verileri de temizleyip yuvarlayalım clean_products = [] for p in products_db: # İsmi hala "Yeni Ürün" olanları canlı kazıyarak kurtarmayı dene if p[1] == "Yeni Ürün": canli_isim, canli_fiyat = sitemizden_veri_kaziyici(p[2]) p[1] = canli_isim if "Bilinmeyen" not in canli_isim else p[1] p[4] = canli_fiyat if canli_fiyat > 0 else p[4] # Kesin yuvarlama işlemi (4.989999999999998 -> 4.99) p[4] = round(float(p[4]), 2) if p[4] < 0: p[4] = abs(p[4]) # Eksili fiyatı pozitife çevir clean_products.append(p) return render_template('index.html', products=clean_products) @app.route('/add', methods=['POST']) def add_product(): url = request.form.get('url') target_price = float(request.form.get('target_price', 0)) # Canlı veriyi çek product_name, current_price = sitemizden_veri_kaziyici(url) # Yapay zeka durum belirteci (Örnek mantık) trend = "STABIL" if current_price < target_price * 0.8: trend = "ALIM" elif current_price > target_price: trend = "DUSUS" # Yeni ürünü listeye/veritabanına ekle new_id = len(products_db) + 1 products_db.append([new_id, product_name, url, target_price, current_price, trend]) return render_template('index.html', products=products_db) @app.route('/api/chart/') def chart_data(product_id): # Grafik için mockup geçmiş verisi üretimi (Küsuratsız temiz sayılar) labels = ["10 Haz", "11 Haz", "12 Haz", "13 Haz", "14 Haz", "Bugün"] # İlgili ürünü bul prod = next((p for p in products_db if p[0] == product_id), None) base_price = prod[4] if prod else 100.0 # Geçmiş fiyat dalgalanmasını temiz yuvarlanmış sayılarla yap prices = [round(base_price * random.uniform(0.95, 1.05), 2) for _ in range(5)] prices.append(base_price) return jsonify({"labels": labels, "prices": prices}) if __name__ == '__main__': app.run(debug=True)
